        // Funá?o para formatar valores monet†rios
        function formatCurrency(input) {
            let value = input.value.replace(/\D/g, ''); // Remove tudo que n?o Ç d°gito
            
            if (value === '') {
                input.value = '';
                return;
            }
            
            // Converte para n£mero e formata
            value = parseInt(value);
            
            // Formata como moeda brasileira
            const formatted = new Intl.NumberFormat('pt-BR', {
                style: 'currency',
                currency: 'BRL',
                minimumFractionDigits: 2
            }).format(value / 100);
            
            input.value = formatted;
        }

        // Funá?o para extrair valor numÇrico de campo formatado
        function getNumericValue(input) {
            if (!input.value) return 0;

                .replace(/\./g, '') // Remove pontos (milhares)

            return parseFloat(numericValue) || 0;
        }

        function resetCalculator() {
            // Limpar todos os campos
            document.querySelectorAll('input').forEach(input => {
                if (input.type === 'checkbox' || input.type === 'radio') {
                    input.checked = false;
                } else {
                    input.value = '';
                }
            });

            // Limpar selects
            document.querySelectorAll('select').forEach(select => {
                select.value = '';
            });

            // Remover todos os alertas
            document.querySelectorAll('.alert').forEach(alert => {
                alert.remove();
            });

            // Esconder campos condicionais
            document.getElementById('meses-funcionamento').classList.add('hidden');
            document.getElementById('valor-prolabore-section').classList.add('hidden');
            document.getElementById('folha-pagamento-section').classList.add('hidden');

            hideAllScreens();
            document.getElementById('screen-initial').classList.remove('hidden');
        }

        function hideAllScreens() {
            const screens = [
                'screen-initial', 'screen-not-eligible', 'screen-alert', 
                'screen-mais12', 'screen-detailed', 'result-anexo3', 'result-anexo5', 'screen-error'
            ];
            
            screens.forEach(screenId => {
                const element = document.getElementById(screenId);
                if (element) {
                    element.classList.add('hidden');
                }
            });
        }
